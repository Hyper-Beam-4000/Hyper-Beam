\section*{2019 USAMO Problems/Problem 5}
Source: AoPS Wiki

\subsection*{Problem}
\frac{m}{n}

\frac{n}{m}

\frac{x+y}{2}

\frac{2xy}{x+y}

(m,n)

\subsection*{Solution / Answer}
m, n

m+n=2^k

\frac{n(\frac{m}{n})+(2^k-n)\frac{n}{m}}{2^k}=\frac{m+n}{2^k}=1

m+n\ne 2^k

m+n\equiv 0\pmod{p}

m\equiv -n\pmod{p}

\frac{m}{n}\equiv \frac{n}{m}\equiv -1\pmod{p}

\frac{-1+(-1)}{2}\equiv -1\pmod{p}

\frac{2(-1)(-1)}{-1+(-1)}\equiv -1\pmod{p}

1\equiv -1\pmod{p}

2\equiv 0\pmod{p}
